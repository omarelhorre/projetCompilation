begin
  // declarations
  int i
  int n
  int somme

  n := 5
  somme := 0

  // boucle for + arithmetique
  for i := 1 to n do
    somme := somme + i * (2 + 1) / 3
  done

  // condition if / else
  if (somme >= 10) then
    write somme
  else
    write 0
  fi

  // boucle while + puissance
  while (n > 0) do
    write 2 ^ n
    n := n - 1
  od

  read(i)
end
